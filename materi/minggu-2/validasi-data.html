Validasi Dasar

validasi.php 
<?php
$errors = [];

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    $nama = trim($_POST['nama']);
    $email = trim($_POST['email']);
    $umur = $_POST['umur'];
    
    // Cek apakah kosong
    if (empty($nama)) {
        $errors[] = "Nama wajib diisi!";
    }
    
    // Validasi email
    if (!filter_var($email, FILTER_VALIDATE_EMAIL)) {
        $errors[] = "Format email tidak valid!";
    }
    
    // Validasi angka
    if (!is_numeric($umur) || $umur < 1) {
        $errors[] = "Umur harus berupa angka positif!";
    }
    
    // Jika tidak ada error
    if (empty($errors)) {
        echo "Data valid! Siap diproses.";
    }
}
?>

Membersihkan Data (Sanitasi)

sanitasi.php 
<?php
// htmlspecialchars - mencegah XSS
$nama = htmlspecialchars($_POST['nama']);

// Menghapus spasi berlebih
$nama = trim($nama);

// Filter email
$email = filter_var($_POST['email'], FILTER_SANITIZE_EMAIL);

// Filter angka saja
$umur = filter_var($_POST['umur'], FILTER_SANITIZE_NUMBER_INT);

// Contoh penggunaan lengkap
function bersihkan($data) {
    $data = trim($data);
    $data = stripslashes($data);
    $data = htmlspecialchars($data);
    return $data;
}

$nama = bersihkan($_POST['nama']);
?>
