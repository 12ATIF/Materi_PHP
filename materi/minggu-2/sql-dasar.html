üìñ Bagian 2: SELECT (Membaca Data)

SQL - SELECT 
-- 1. Ambil SEMUA data, SEMUA kolom
SELECT * FROM siswa;

-- 2. Ambil kolom TERTENTU saja
SELECT nama, kelas FROM siswa;

-- 3. Filter dengan WHERE
SELECT * FROM siswa WHERE kelas = 'X RPL 1';
SELECT * FROM siswa WHERE umur > 15;
SELECT * FROM siswa WHERE nama LIKE '%Budi%';  -- Berisi "Budi"

-- 4. Urutkan dengan ORDER BY
SELECT * FROM siswa ORDER BY nama ASC;   -- A-Z
SELECT * FROM siswa ORDER BY nama DESC;  -- Z-A
SELECT * FROM siswa ORDER BY id DESC;    -- Terbaru dulu

-- 5. Batasi jumlah dengan LIMIT
SELECT * FROM siswa LIMIT 10;           -- Ambil 10 data pertama
SELECT * FROM siswa LIMIT 5, 10;        -- Skip 5, ambil 10 (pagination)

-- 6. Kombinasi lengkap
SELECT nama, kelas, email 
FROM siswa 
WHERE kelas = 'X RPL 1' 
ORDER BY nama ASC 
LIMIT 10;

üìñ Operator WHERE yang Sering Digunakan:

Operator Arti Contoh
= Sama dengan WHERE kelas = 'X RPL 1'
!= Tidak sama WHERE kelas != 'X RPL 1'
> , << /td> Lebih/kurang dari WHERE umur > 15
LIKE Cocok pola WHERE nama LIKE '%Budi%'
IN Dalam list WHERE kelas IN ('X RPL 1', 'X RPL 2')
AND, OR Gabungan WHERE kelas = 'X RPL 1' AND umur > 15

‚ûï Bagian 3: INSERT (Menambah Data)

SQL - INSERT 
-- Sintaks dasar
INSERT INTO nama_tabel (kolom1, kolom2, kolom3) 
VALUES ('nilai1', 'nilai2', 'nilai3');

-- Contoh: Tambah siswa baru
INSERT INTO siswa (nama, kelas, email) 
VALUES ('Budi Santoso', 'X RPL 1', 'budi@email.com');

-- Tambah banyak data sekaligus
INSERT INTO siswa (nama, kelas, email) VALUES 
('Ani Wijaya', 'X RPL 1', 'ani@email.com'),
('Citra Dewi', 'X RPL 2', 'citra@email.com'),
('Doni Prasetyo', 'XI RPL 1', 'doni@email.com');
üí°
Tips: Kolom id dengan AUTO_INCREMENT tidak perlu diisi. MySQL akan otomatis memberikan nomor urut!

‚úèÔ∏è Bagian 4: UPDATE (Mengubah Data)

SQL - UPDATE 
-- Sintaks dasar
UPDATE nama_tabel 
SET kolom1 = 'nilai_baru' 
WHERE kondisi;

-- Contoh: Ubah kelas siswa dengan id = 1
UPDATE siswa 
SET kelas = 'XI RPL 1' 
WHERE id = 1;

-- Ubah banyak kolom sekaligus
UPDATE siswa 
SET kelas = 'XI RPL 1', email = 'budi.baru@email.com' 
WHERE id = 1;

-- Update banyak data sekaligus
UPDATE siswa 
SET kelas = 'XI RPL 1' 
WHERE kelas = 'X RPL 1';  -- Semua siswa X RPL 1 naik kelas
‚ö†Ô∏è
BAHAYA! SELALU gunakan WHERE! Tanpa WHERE, SEMUA data akan terubah!
‚ùå UPDATE siswa SET kelas = 'X'; ‚Üí Semua siswa jadi kelas X!

üóëÔ∏è Bagian 5: DELETE (Menghapus Data)

SQL - DELETE 
-- Sintaks dasar
DELETE FROM nama_tabel WHERE kondisi;

-- Contoh: Hapus siswa dengan id = 5
DELETE FROM siswa WHERE id = 5;

-- Hapus siswa dengan kondisi tertentu
DELETE FROM siswa WHERE kelas = 'X RPL 1' AND status = 'nonaktif';

-- ‚ö†Ô∏è HATI-HATI! Ini menghapus SEMUA data!
DELETE FROM siswa;  -- JANGAN LAKUKAN INI!
‚ö†Ô∏è
SANGAT BAHAYA! DELETE FROM siswa; tanpa WHERE akan menghapus SEMUA data dalam tabel! Data yang dihapus TIDAK BISA dikembalikan!

üíª Bagian 6: Menjalankan SQL di PHP

SELECT dengan mysqli

tampil_data.php 
<?php
require_once 'koneksi.php';

// Query SELECT
$sql = "SELECT * FROM siswa ORDER BY id DESC";
$result = mysqli_query($conn, $sql);

// Cek apakah ada data
if (mysqli_num_rows($result) > 0) {
    // Loop setiap baris data
    while ($row = mysqli_fetch_assoc($result)) {
        echo "ID: " . $row['id'] . "<br>";
        echo "Nama: " . $row['nama'] . "<br>";
        echo "Kelas: " . $row['kelas'] . "<br>";
        echo "<hr>";
    }
} else {
    echo "Tidak ada data siswa.";
}

// Tutup koneksi
mysqli_close($conn);
?>

INSERT dengan mysqli

tambah_data.php 
<?php
require_once 'koneksi.php';

if ($_SERVER['REQUEST_METHOD'] == 'POST') {
    // Ambil & sanitasi data dari form
    $nama = mysqli_real_escape_string($conn, $_POST['nama']);
    $kelas = mysqli_real_escape_string($conn, $_POST['kelas']);
    $email = mysqli_real_escape_string($conn, $_POST['email']);
    
    // Query INSERT
    $sql = "INSERT INTO siswa (nama, kelas, email) 
            VALUES ('$nama', '$kelas', '$email')";
    
    // Jalankan query
    if (mysqli_query($conn, $sql)) {
        echo "‚úÖ Data berhasil ditambahkan!";
    } else {
        echo "‚ùå Error: " . mysqli_error($conn);
    }
}

mysqli_close($conn);
?>

üìñ Fungsi mysqli Penting:

Fungsi Kegunaan
mysqli_query($conn, $sql) Jalankan query SQL
mysqli_fetch_assoc($result) Ambil 1 baris data sebagai array assoc
mysqli_num_rows($result) Hitung jumlah baris hasil
mysqli_real_escape_string() Escape string (cegah SQL Injection)
mysqli_error($conn) Tampilkan pesan error
mysqli_close($conn) Tutup koneksi
