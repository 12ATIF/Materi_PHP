üìñ Perintah SQL Dasar

1. SELECT - Mengambil Data

SQL 
-- Ambil semua data
SELECT * FROM siswa;

-- Ambil kolom tertentu
SELECT nama, kelas FROM siswa;

-- Dengan filter
SELECT * FROM siswa WHERE kelas = 'X RPL 1';

-- Urutkan
SELECT * FROM siswa ORDER BY nama ASC;

2. INSERT - Menambah Data

SQL 
INSERT INTO siswa (nama, kelas, email) 
VALUES ('Budi Santoso', 'X RPL 1', 'budi@email.com');

3. UPDATE - Mengubah Data

SQL 
UPDATE siswa SET kelas = 'XI RPL 1' WHERE id = 1;

4. DELETE - Menghapus Data

SQL 
DELETE FROM siswa WHERE id = 1;
